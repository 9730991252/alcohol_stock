************** Create System Service File for Gunicorn ******
Syntax:- sudo nano /etc/systemd/system/your_domain.gunicorn.service
Example:- sudo nano /etc/systemd/system/narayaniorder.com.gunicorn.service

cd  /etc/systemd/system/

sudo nano alcohol_stock.com.gunicorn.service


[Unit]
Description=alcohol_stock.com.gunicorn daemon
Requires=alcohol_stock.com.gunicorn.socket
After=network.target

[Service]
User=root
Group=root
WorkingDirectory=/root/alcohol_stock
ExecStart=/root/alcohol_stock/venv/bin/gunicorn \
          --access-logfile - \
          --workers 3 \
          --bind unix:/run/alcohol_stock.com.gunicorn.sock \
          alcohol_stock.wsgi:application

[Install]
WantedBy=multi-user.target
